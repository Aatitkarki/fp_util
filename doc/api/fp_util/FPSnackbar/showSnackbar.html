Implementation

static showSnackbar(
  BuildContext context, {
  required String message,
  int duration = 2,
  Color backgroundColor = const Color(0xFF1C1C1C),
  TextStyle textStyle =
      const TextStyle(fontSize: 18, color: Color(0xFFFBFBFB)),
  double borderRadius = 10.0,
  Border? border,
  Widget? icon,
}) {
  final snackbarContent = Text(
    message,
    style: textStyle,
  );
  ScaffoldMessenger.of(context)
    ..hideCurrentSnackBar()
    ..showSnackBar(
      SnackBar(
        content: Container(
          width: context.width(),
          constraints: const BoxConstraints(minHeight: Sizes.p54),
          decoration: BoxDecoration(
            color: backgroundColor,
            borderRadius: borderRadius.circular(),
            border: border,
          ),
          margin: const EdgeInsets.symmetric(horizontal: Sizes.p16),
          padding: const EdgeInsets.all(Sizes.p12),
          child: icon != null
              ? Row(
                  crossAxisAlignment: CrossAxisAlignment.center,
                  children: [
                    icon,
                    Sizes.gapH12(),
                    Expanded(child: snackbarContent),
                  ],
                )
              : snackbarContent,
        ),
        behavior: SnackBarBehavior.floating,
        shape: RoundedRectangleBorder(borderRadius: borderRadius.circular()),
        backgroundColor: Colors.white,
        padding: EdgeInsets.zero,
        elevation: 0,
        clipBehavior: Clip.antiAlias,
        duration: Duration(seconds: duration),
        margin: const EdgeInsets.symmetric(
            horizontal: Sizes.p32, vertical: Sizes.p24),
      ),
    );
}

# Num Extensions

### Usage
Add the following import to your Dart file:
```dart
import 'package:fp_util/fp_util.dart';
```

### gap
```dart title="gap widget"
import 'package:flutter/material.dart';
import 'package:fp_util/fp_util.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        body: Center(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: <Widget>[
              Text('Hello World'),
              gap(10),
              Text('Hello World'),
            ],
          ),
        ),
      ),
    );
  }
}
```

### sliverGap
```dart title="sliverGap widget"
import 'package:flutter/material.dart';
import 'package:fp_util/fp_util.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        body: CustomScrollView(
          slivers: <Widget>[
            SliverList(
              delegate: SliverChildListDelegate(
                [
                  Text('Hello World'),
                  10.sliverGap,
                  Text('Hello World'),
                ],
              ),
            ),
          ],
        ),
      ),
    );
  }
}
```

### microseconds
```dart title="get duration on microseconds"
  final duration = 1.microseconds;
  print(duration); // 0:00:00.001000
```

### milliseconds
